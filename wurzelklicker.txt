Shinies
	*LadyinRed
		on click:log Woop
		// Starte rechts vom Bildschirm, bewege dich nach links, fade in und out, bewege dich hoch und runter mit Geschwindigkeit 0.05
		movement:onRight moveLeft fade bounce:0.05
		// Dauer bis zum Erscheinen mindestens (sec)
		frequency:1
		// Zeitspanne bis zum Erscheinen höchstens (= frequency + frequency variation) (sec)
		frequency variation:59
		// Dauer, die die Grafik angezeigt wird (sec)
		duration: 10
		// Grafik. Quelle: https://www.publicdomainpictures.net/en/view-image.php?image=473923&picture=red-dress-woman-evening-dress
		icon:https://raw.githubusercontent.com/aaaaaprvdgrwwelt/Wurzelklicker/main/grafiken/lady-in-red2.png
		class:bigButton
		on click:
			$boost=1
			if (have clover) $boost=4
			if (chance(25%))
				//get at least 7, or between 1 and 3 minutes of our taler production
				$amount=max(7,random(taler:ps*60*1,taler:ps*60*3))*$boost
				toast Die Dame in Rot schenkt dir<//><b>[$amount] Taler</b>.
				yield $amount taler
			else
				$amount=1*$boost
				toast Die Dame in Rot schenkt dir<//><b>[$amount] goldene Pommes</b>!
				yield $amount GoldenePommes
			end
		end
